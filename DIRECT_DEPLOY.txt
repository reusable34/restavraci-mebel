ВЫПОЛНИТЕ ЭТУ КОМАНДУ НА СЕРВЕРЕ (root@kolas:~#):

apt-get update && apt-get install -y curl wget build-essential && \
cd /tmp && \
wget -q http://192.168.0.112:8080/restavraci-deploy.tar.gz -O restavraci.tar.gz && \
mkdir -p /opt/restavraci && \
cd /opt/restavraci && \
tar -xzf /tmp/restavraci.tar.gz && \
curl -fsSL https://deb.nodesource.com/setup_20.x | bash - && \
apt-get install -y nodejs && \
npm ci && \
npm run build && \
cat > .env << 'EOF'
SANITY_PROJECT_ID=143zykun
SANITY_DATASET=production
SANITY_API_VERSION=2025-01-01
TELEGRAM_BOT_TOKEN=your_telegram_bot_token_here
TELEGRAM_CHAT_ID=your_telegram_chat_id_here
CONTACT_EMAIL=noreply@example.com
NEXT_PUBLIC_SANITY_PROJECT_ID=143zykun
NEXT_PUBLIC_SANITY_DATASET=production
NEXT_PUBLIC_SANITY_API_VERSION=2025-01-01
PORT=3000
EOF
cat > /etc/systemd/system/restavraci.service << 'EOF'
[Unit]
Description=RestavraciMebel Next.js Application
After=network.target
[Service]
Type=simple
User=root
WorkingDirectory=/opt/restavraci
Environment="NODE_ENV=production"
EnvironmentFile=/opt/restavraci/.env
ExecStart=/usr/bin/npm start
Restart=always
RestartSec=10
StandardOutput=journal
StandardError=journal
[Install]
WantedBy=multi-user.target
EOF
systemctl daemon-reload && \
systemctl enable restavraci.service && \
systemctl start restavraci.service && \
sleep 2 && \
systemctl status restavraci.service && \
echo "✅ Деплой завершен! Настройте Telegram: nano /opt/restavraci/.env"
